az monitor log-analytics workspace table create \
  --resource-group virtualmachines-rg \
  --workspace-name lgws-vm \
  --name ApacheAccess_CL \
  --columns \
    TimeGenerated=datetime \
    ClientIP=string \
    RequestTime=datetime \
    Method=string \
    UriStem=string \
    HttpVersion=string \
    StatusCode=int \
    BytesSent=int \
    Referer=string \
    UserAgent=string




    
PORT SCAN:
// Real external traffic only, parse key fields
Syslog
| where TimeGenerated > ago(1h)
| where Facility == "kern"
| where SyslogMessage contains "IPTABLES-IN"
| where SyslogMessage !contains "IN=lo"  // exclude loopback noise
| extend SrcIP = extract(@"SRC=([\d\.]+)", 1, SyslogMessage)
| extend DstPort = extract(@"DPT=(\d+)", 1, SyslogMessage)
| extend Proto = extract(@"PROTO=(\w+)", 1, SyslogMessage)
| extend InIface = extract(@"IN=(\w+)", 1, SyslogMessage)
| summarize PortsScanned = dcount(DstPort), Attempts = count(), Ports = make_set(DstPort) 
    by SrcIP, Proto, bin(TimeGenerated, 5m)
| where PortsScanned > 3
| order by Attempts desc
